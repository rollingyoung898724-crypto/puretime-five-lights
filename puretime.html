import React, { useMemo, useState } from "react";


export default function PureTimeFiveLightsMVP() {

  const [tab, setTab] = useState("today");

  const [todayStars, setTodayStars] = useState([

    { name: "Fajr", active: true },

    { name: "Dhuhr", active: true },

    { name: "Asr", active: false },

    { name: "Maghrib", active: true },

    { name: "Isha", active: false },

  ]);


  const pathDays = [

    { day: 1, lights: 5 },

    { day: 2, lights: 4 },

    { day: 3, lights: 5 },

    { day: 4, lights: 2 },

    { day: 5, today: true },

    { day: 6, lights: 0 },

    { day: 7, lights: 0 },

  ];


  const activeCount = useMemo(

    () => todayStars.filter((star) => star.active).length,

    [todayStars]

  );


  const totalLights = useMemo(() => {

    const previous = pathDays.reduce((sum, item) => sum + (item.today ? 0 : item.lights), 0);

    return previous + activeCount;

  }, [activeCount]);


  const currentPathDays = useMemo(() => {

    return pathDays.map((item) => (item.today ? { ...item, lights: activeCount } : item));

  }, [activeCount]);


  const glowClass = (n) => {

    if (n === 5) return "bg-amber-100 shadow-[0_0_18px_rgba(255,240,180,0.65)]";

    if (n >= 3) return "bg-amber-200/90 shadow-[0_0_14px_rgba(255,230,160,0.45)]";

    if (n >= 1) return "bg-amber-300/70 shadow-[0_0_10px_rgba(255,220,140,0.28)]";

    return "bg-white/10";

  };


  const toggleStar = (index) => {

    setTodayStars((prev) =>

      prev.map((star, i) => (i === index ? { ...star, active: !star.active } : star))

    );

  };


  const TabButton = ({ id, label }) => (

    <button

      onClick={() => setTab(id)}

      className={`flex-1 rounded-2xl px-3 py-3 text-sm font-medium transition ${

        tab === id

          ? "bg-white/14 text-white"

          : "bg-transparent text-slate-400 hover:text-slate-200"

      }`}

    >

      {label}

    </button>

  );


  const TodayView = () => (

    <section className="rounded-[28px] border border-white/10 bg-white/5 p-5 shadow-2xl backdrop-blur">

      <div className="mb-6">

        <p className="text-xs uppercase tracking-[0.2em] text-slate-500">Today</p>

        <h1 className="mt-2 text-3xl font-semibold">April 5</h1>

        <p className="mt-2 text-sm text-slate-400">A little light still counts.</p>

      </div>


      <div className="space-y-3">

        {todayStars.map((star, index) => (

          <button

            key={star.name}

            onClick={() => toggleStar(index)}

            className="flex w-full items-center justify-between rounded-2xl border border-white/10 bg-slate-900/60 px-4 py-4 text-left transition active:scale-[0.99]"

          >

            <div>

              <p className="text-sm font-medium">{star.name}</p>

              <p className="mt-1 text-xs text-slate-500">

                {star.active ? "Light recorded" : "Tap to light"}

              </p>

            </div>

            <div

              className={`h-5 w-5 rounded-full transition ${

                star.active

                  ? "bg-amber-100 shadow-[0_0_16px_rgba(255,235,170,0.75)]"

                  : "bg-white/15"

              }`}

            />

          </button>

        ))}

      </div>


      <div className="mt-7 rounded-[28px] border border-white/10 bg-slate-900/70 p-5">

        <div className="mb-4 flex items-center justify-between">

          <p className="text-sm text-slate-400">Today Node</p>

          <p className="text-sm text-slate-300">{activeCount} / 5</p>

        </div>

        <div className="flex items-center justify-center py-5">

          <div

            className={`relative rounded-full transition-all duration-300 ${

              activeCount === 0

                ? "h-16 w-16 bg-white/10"

                : activeCount === 1

                ? "h-20 w-20 bg-amber-300/60 shadow-[0_0_18px_rgba(255,220,140,0.18)]"

                : activeCount === 2

                ? "h-24 w-24 bg-amber-200/75 shadow-[0_0_22px_rgba(255,230,160,0.22)]"

                : activeCount === 3

                ? "h-28 w-28 bg-amber-100/85 shadow-[0_0_30px_rgba(255,230,160,0.32)]"

                : activeCount === 4

                ? "h-32 w-32 bg-amber-100/90 shadow-[0_0_38px_rgba(255,235,170,0.4)]"

                : "h-36 w-36 bg-amber-50 shadow-[0_0_54px_rgba(255,240,180,0.5)]"

            }`}

          >

            <div className="absolute inset-3 rounded-full border border-white/15" />

            <div className="absolute inset-7 rounded-full border border-white/10" />

          </div>

        </div>

        <p className="text-center text-sm text-slate-300">

          {activeCount === 5 ? "Today is fully lit." : `${activeCount} of 5 lights today`}

        </p>

      </div>


      <button

        onClick={() => setTab("path")}

        className="mt-6 w-full rounded-2xl border border-white/10 bg-white/10 py-4 text-sm font-medium transition hover:bg-white/15"

      >

        View My Path

      </button>

    </section>

  );


  const PathView = () => (

    <section className="rounded-[28px] border border-white/10 bg-white/5 p-5 shadow-2xl backdrop-blur">

      <div className="mb-6">

        <p className="text-xs uppercase tracking-[0.2em] text-slate-500">Star Path</p>

        <h2 className="mt-2 text-3xl font-semibold">April Path</h2>

        <div className="mt-4 flex flex-wrap gap-2 text-xs text-slate-300">

          <span className="rounded-full bg-white/8 px-3 py-1.5">5 days lit</span>

          <span className="rounded-full bg-white/8 px-3 py-1.5">{totalLights} lights</span>

          <span className="rounded-full bg-white/8 px-3 py-1.5">3-day streak</span>

        </div>

      </div>


      <div className="relative ml-2 space-y-5 before:absolute before:left-[7px] before:top-2 before:h-[calc(100%-16px)] before:w-px before:bg-white/10">

        {currentPathDays.map((item) => (

          <div key={item.day} className="relative flex items-start gap-4">

            <div

              className={`relative z-10 mt-1 h-4 w-4 rounded-full ${glowClass(item.lights)} ${

                item.today ? "ring-2 ring-amber-100/40" : ""

              }`}

            />

            <div className="flex-1 rounded-2xl border border-white/10 bg-slate-900/50 px-4 py-3.5">

              <div className="flex items-center justify-between">

                <p className="text-sm font-medium">Day {item.day}</p>

                <p className="text-xs text-slate-400">{item.lights}/5</p>

              </div>

              <div className="mt-3 flex gap-2">

                {Array.from({ length: 5 }).map((_, i) => (

                  <div

                    key={i}

                    className={`h-2.5 flex-1 rounded-full ${

                      i < item.lights ? "bg-amber-100/90" : "bg-white/10"

                    }`}

                  />

                ))}

              </div>

            </div>

          </div>

        ))}

      </div>


      <button

        onClick={() => setTab("reflection")}

        className="mt-6 w-full rounded-2xl border border-white/10 bg-white/10 py-4 text-sm font-medium transition hover:bg-white/15"

      >

        Monthly Reflection

      </button>

    </section>

  );


  const ReflectionView = () => (

    <section className="rounded-[28px] border border-white/10 bg-white/5 p-5 shadow-2xl backdrop-blur">

      <div className="mb-6">

        <p className="text-xs uppercase tracking-[0.2em] text-slate-500">Reflection</p>

        <h2 className="mt-2 text-3xl font-semibold">April Summary</h2>

        <p className="mt-2 text-sm text-slate-400">

          Not every day was full, but the path is still yours.

        </p>

      </div>


      <div className="rounded-[28px] border border-white/10 bg-slate-900/60 p-5">

        <div className="grid grid-cols-7 gap-3">

          {Array.from({ length: 28 }).map((_, i) => {

            const filled = i < totalLights;

            return (

              <div

                key={i}

                className={`aspect-square rounded-full ${

                  filled

                    ? "bg-amber-100/85 shadow-[0_0_12px_rgba(255,235,170,0.35)]"

                    : "bg-white/8"

                }`}

              />

            );

          })}

        </div>

      </div>


      <div className="mt-6 grid grid-cols-3 gap-3 text-center">

        <div className="rounded-2xl border border-white/10 bg-slate-900/50 p-4">

          <p className="text-xl font-semibold">5</p>

          <p className="mt-1 text-xs text-slate-400">Days Lit</p>

        </div>

        <div className="rounded-2xl border border-white/10 bg-slate-900/50 p-4">

          <p className="text-xl font-semibold">{totalLights}</p>

          <p className="mt-1 text-xs text-slate-400">Total Lights</p>

        </div>

        <div className="rounded-2xl border border-white/10 bg-slate-900/50 p-4">

          <p className="text-xl font-semibold">3</p>

          <p className="mt-1 text-xs text-slate-400">Best Streak</p>

        </div>

      </div>

    </section>

  );


  return (

    <div className="min-h-screen bg-slate-950 text-slate-100">

      <div className="mx-auto flex min-h-screen w-full max-w-md flex-col px-4 pb-28 pt-6">

        <div className="mb-6 flex items-center justify-between">

          <div>

            <p className="text-xs uppercase tracking-[0.25em] text-slate-500">PureTime</p>

            <p className="mt-1 text-lg font-medium text-slate-200">Five Lights</p>

          </div>

          <div className="rounded-full border border-white/10 bg-white/5 px-3 py-1.5 text-xs text-slate-400">

            Gentle Mode

          </div>

        </div>


        <div className="flex-1">

          {tab === "today" && <TodayView />}

          {tab === "path" && <PathView />}

          {tab === "reflection" && <ReflectionView />}

        </div>


        <nav className="fixed bottom-0 left-0 right-0 border-t border-white/10 bg-slate-950/90 px-4 py-4 backdrop-blur">

          <div className="mx-auto flex max-w-md rounded-3xl border border-white/10 bg-white/5 p-1.5">

            <TabButton id="today" label="Today" />

            <TabButton id="path" label="Path" />

            <TabButton id="reflection" label="Reflection" />

          </div>

        </nav>

      </div>

    </div>

  );

}
